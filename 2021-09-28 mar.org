function sums (a,f){
    let b = []
    let acc = a[0]
    b.push(acc)
    for (i = 1; i<a.length; i++){
        acc = f(acc,a[i])
        b.push(acc)
    }
    return b
}

let array = [2,3,4,1];
console.log(sums(array,(a,b)=>a*b))
